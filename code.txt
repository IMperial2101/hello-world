#include <stdio.h>
#include <iostream>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Russian");
	int x = 21;
	double y = 21.21;
	int a, b, f, p;
	a = 9;
	b = 6;
	f = x + y;
	double z = (double)a / b; // получение дробного числа путем деления двух чисел типа integer
	double k = 45.453245353;
	p = (int)k; // Приведение вещественного типа к целочисленному
	cout << p << endl;
	cout << z << endl;
	cout << f << endl; // сложение двух типов чисел(вещественного и целочисленного)(неявное приведение одного типа данных к другому)

	cout << "Введите два числа для сравнения" << endl;

	float sim1, two1;
	cin >> sim1;       // ввод двух чисел пользователем 
	cin >> two1;

	cout << (sim1 >= two1 ? sim1 : two1) * 4 + z + (int)k - f << endl;
	// 1.Вычисление алгебраического выражения, содержащего операции различного уровня приоритетности Задается преподавателем
	// 2.	Вычисление алгебраического выражения, требующего преобразования типов. Задается преподавателем

	system("pause"); // если командная строка закрывается сразу 
	return 0;

}
